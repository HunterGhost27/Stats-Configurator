//  |                                                      MOD INITIALIZATION                                                          |

KBSECTION

//  ===================
//  STARTING A NEW GAME
//  ===================

IF
GameEventSet("GAMEEVENT_GameStarted")
THEN
GlobalSetFlag("S7_Config_ModAddedTo_NewGame");
GlobalSetFlag("S7_ConfigActive");
NRD_ModCall("S7_Config", "S7_ConfigLog", "Parent Goal Initialized. Mod added to a new game.", "[Initializer]", "ModAddedTo", "NewGame");
GoalCompleted;

//  =============================
//  LOADING AN EXISTING SAVE-GAME
//  =============================

IF
SavegameLoaded(_, _, _, _)
AND
DB_StoryStarted(_)
THEN
GlobalSetFlag("S7_ConfigActive");
NRD_ModCall("S7_Config", "S7_ConfigLog", "Parent Goal Initialized. Story was already initialized. Mod added to an existing save-file.", "[Initializer]", "ModAddedTo", "ExistingSave");
GoalCompleted;

EXITSECTION

ENDEXITSECTION
ParentTargetEdge "__Start"
